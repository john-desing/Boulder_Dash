20  public class BoulderThread extends Thread {
21  	/** The View */
22  	IView view;
23  	/** The Model */
24  	IModel model;
25  
26  	/** Contains the position X and Y of the Boulder */
27  	int[][] tabBoulder = new int[2][210];
28  	/** The Entity Array */
29  	Entity[][] tabEntity;
30  	/** The Sprite Array */
31  	Image[][] tabMap;
32  
33  	/** The Boulder */
34  	Boulder boulder;
35  	/** The Diamond */
36  	Diamond diamond;
37  
38  	/**
39  	 * Instantiates a new BoulderThread
40  	 * 
41  	 * @param name  The name of the Thread
42  	 * @param view  The View
43  	 * @param model The Model
44  	 */
45  	public BoulderThread(String name, IView view, IModel model) {
46  		super(name);
47  		this.view = view;
48  		this.model = model;
49  	}
50  
51  	/**
52  	 * Automatically executed at the beginning of the Thread, Execute the function
53  	 * fall() every 100 Ms
54  	 */
55  	public void run() {
56  		while (true)
57  			try {
58  				this.fall();
59  				Thread.sleep(10);
60  			} catch (Exception e) {
61  				e.printStackTrace();
62  			}
63  	}
64  
65  	/**
66  	 * Swap is used to make boulders falls
67  	 * 
68  	 * @param i The boulder's X position before the swap
69  	 * @param j The boulder's Y position before the swap
70  	 */
71  	public void swap(int i, int j) {
72  		Entity entity;
73  		entity = tabEntity[j][i];
74  		tabEntity[j][i] = tabEntity[j + 1][i];
75  		tabEntity[j + 1][i] = entity;
76  		Image image;
77  		image = tabMap[j][i];
78  		tabMap[j][i] = tabMap[j + 1][i];
79  		tabMap[j + 1][i] = image;
80  		model.update();
81  	}
82  
83  	/**
84  	 * Load the position X and Y of all the boulders in the TabEntity
85  	 * 
86  	 * @return An array with all the boulders' X and Y position
87  	 */
88  	public int[][] definition() {
89  		this.tabEntity = model.getTabEntity();
90  		this.tabMap = model.getMap();
91  		int k = 0;
92  		for (int i = 0; i < 22; i++) {
93  			for (int j = 0; j < 40; j++) {
94  				if (tabEntity[i][j] instanceof Boulder || tabEntity[i][j] instanceof Diamond) {
95  					tabBoulder[0][k] = j;
96  					tabBoulder[1][k] = i;
97  					k++;
98  					tabEntity[i][j].setFalling(false);
99  					tabEntity[i][j].setCanBePushed(false);
100 				}
101 			}
102 		}
103 		return tabBoulder;
104 	}
105 
106 	/**
107 	 * Check if the Boulders/Dimaonds can fall, if the player is under the boulder
108 	 * or diamond, if the boulder can be pushed and reload the Level if the player
109 	 * dies.
110 	 */
111 	public void fall() {
112 		tabBoulder = definition();
113 		int j = 0;
114 		for (j = 0; j < 210; j++) {
115 			if (tabEntity[tabBoulder[1][j] + 1][tabBoulder[0][j]] instanceof Air
116 					|| tabEntity[tabBoulder[1][j] + 1][tabBoulder[0][j]] instanceof Character) {
117 				if (tabEntity[tabBoulder[1][j] + 1][tabBoulder[0][j]] instanceof Air) {
118 					swap(tabBoulder[0][j], tabBoulder[1][j]);
119 					tabEntity[tabBoulder[1][j] + 1][tabBoulder[0][j]].setFalling(true);
120 					if (tabEntity[tabBoulder[1][j] + 1][tabBoulder[0][j]].getFalling()
121 							&& tabEntity[tabBoulder[1][j] + 2][tabBoulder[0][j]] instanceof Character) {
122 						view.printMessage("              Game Over\n                Try Again");
123 						model.loadMap(model.getLevel());
124 					}
125 				}
126 			}
127 			if (tabBoulder[0][j] != 0 && tabEntity[tabBoulder[1][j]][tabBoulder[0][j]] instanceof Boulder) {
128 				if (tabEntity[tabBoulder[1][j]][tabBoulder[0][j] - 1] instanceof Air
129 						|| tabEntity[tabBoulder[1][j]][tabBoulder[0][j] + 1] instanceof Air) {
130 					tabEntity[tabBoulder[1][j]][tabBoulder[0][j]].setCanBePushed(true);
131 				}
132 			}
133 		}
134 	}
